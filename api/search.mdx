Search for subjects using a hybrid search approach that combines traditional text search with semantic search. Supports filtering by city, person, party, topic, date range, and geographic location.

### Request body

<ParamField body="query" type="string" required>
  The search query text
</ParamField>

<ParamField body="cityIds" type="array">
  Array of city IDs to filter by
</ParamField>

<ParamField body="personIds" type="array">
  Array of person IDs to filter by
</ParamField>

<ParamField body="partyIds" type="array">
  Array of party IDs to filter by
</ParamField>

<ParamField body="topicIds" type="array">
  Array of topic IDs to filter by
</ParamField>

<ParamField body="dateRange" type="object">
  Date range for filtering results
  <Expandable title="properties">
    <ParamField body="start" type="string">
      Start date in ISO 8601 format (e.g., "2024-01-01T00:00:00Z")
    </ParamField>
    <ParamField body="end" type="string">
      End date in ISO 8601 format (e.g., "2024-12-31T23:59:59Z")
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="location" type="object">
  Geographic location filter
  <Expandable title="properties">
    <ParamField body="point" type="object" required>
      Center point coordinates
      <Expandable title="properties">
        <ParamField body="lat" type="number" required>
          Latitude
        </ParamField>
        <ParamField body="lon" type="number" required>
          Longitude
        </ParamField>
      </Expandable>
    </ParamField>
    <ParamField body="radius" type="number" default="5">
      Search radius in kilometers
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="page" type="integer" default="1">
  Page number (minimum: 1)
</ParamField>

<ParamField body="pageSize" type="integer" default="10">
  Results per page (minimum: 1, maximum: 100)
</ParamField>

<ParamField body="detailed" type="boolean" default="false">
  Whether to return detailed results with speaker segment text. When true, includes full `speakerSegments` array in results.
</ParamField>

### Response

<ResponseField name="results" type="array">
  Array of search results
  <Expandable title="SearchResultLight properties">
    <ResponseField name="id" type="string">
      Subject identifier
    </ResponseField>
    <ResponseField name="name" type="string">
      Subject name
    </ResponseField>
    <ResponseField name="description" type="string">

    </ResponseField>
    <ResponseField name="councilMeeting" type="object">

    </ResponseField>
    <ResponseField name="topic" type="object | null">

    <ResponseField name="introducedBy" type="object | null">
      Person who introduced the subject
    </ResponseField>
    <ResponseField name="location" type="object | null">

## Search configuration

The search API uses the following configuration:

- **Semantic search**: Enabled by default, combining text and vector similarity
- **Rank window size**: 100 results are considered for ranking
- **Rank constant**: 60 (controls the balance between text and semantic scores)
- **Maximum page size**: 100 results per page

## Use cases

### Basic text search

```json
{
  "query": "πάρκα",
  "page": 1,
  "pageSize": 20
}
```

### Filter by city and date range

```json
{
  "query": "οικονομικά",
  "cityIds": ["athens"],
  "dateRange": {
    "start": "2024-01-01T00:00:00Z",
    "end": "2024-12-31T23:59:59Z"
  }
}
```

### Geographic search

```json
{
  "query": "ανακαίνιση",
  "location": {
    "point": {
      "lat": 37.9838,
      "lon": 23.7275
    },
    "radius": 10
  }
}
```

### Detailed results with speaker segments

```json
{
  "query": "κυκλοφορία",
  "cityIds": ["athens"],
  "detailed": true,
  "pageSize": 5
}
```
